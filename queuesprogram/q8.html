#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int rear = -1, front = -1;
int priorities[5];
char values[5][30];

int isFull() {
    if (rear == 4) {
        printf("Queue is full");
    } else {
        printf("Queue is not full");
    }
}

int isEmpty() {
    if (front == -1 && rear==-1) 
        printf("Queue is Empty");
    else 
        printf("Queue is not Empty");

}

void enqueue(int priority, char value[30]) {
    if (rear == 4) {
        printf("Queue is full");
    } else {
        rear = rear + 1;
        priorities[rear] = priority;
        strcpy(values[rear],value);
    }
}

int getHighestPriority() {
    int i, priority;
    priority = priorities[0];
    for (i = 1; i <= rear; i++) {
        if (priorities[i] > priority) {
            priority = priorities[i];
        }
    }
    return priority;
}

char* delete() {
    int i, j,priority;

    for (i = 0; i <= rear; i++) {

        break;
      }
    }
    if (i < rear) {
    for (j = i; j < rear; j++) {
    strcpy(values[j],values[j + 1]);
    priorities[j] = priorities[j + 1];
        }
    }
    rear = rear - 1;
    return value;
}

    printf("Elements in queue are: ");
    for (i = 0; i <= rear; i++) {
    printf("\nPriority is: %d\tValue is: %s", priorities[i],values[i]);
    }
}
