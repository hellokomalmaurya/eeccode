#include<stdio.h>
#include<conio.h>
int linearSearch(int arr[],int search, int index){
if(arr[index]=='\0'){
return -1;
}
else if(arr[index]==search){
return index;
}
else{
return linearSearch(arr,search,index+1);
}
}
