//Enter a string from the user, enter a character and perform binary search algorithm through iteration.
#include<stdio.h>
#include<conio.h>
int binarysearch(char str[],int left,int right,char element){
int mid;
while(left<=right){
mid=(left+right)/2;
if(str[mid]==element){
return mid;
}
if(str[mid]<element){
left=mid+1;
}
if(str[mid]>element){
right=mid-1;
}
}
return -1;
}
