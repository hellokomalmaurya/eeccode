//Enter a string from the user, Enter a character and perform binary search algorithm through recusion.
#include<stdio.h>
#include<conio.h>
int binarySearch(int arr[],int left,int right,int search){
int mid;
if(left>right){
return -1;
}
mid=left+(right-left)/2;
if(arr[mid]==search){
return mid;
}
else if(arr[mid]<search){
return binarySearch(arr,mid+1,right,search);
}
else{
return binarySearch(arr,mid-1,left,search);
}
}
