//Enter a integer array from the user, enter an integer value and perform binary search algorithm through recursion.
#include<stdio.h>
#include<conio.h>
binarysearch(int arr[],int left,int right,int element){
int mid;
if(left<=left){
mid=(left+right)/2;
if(arr[mid]==element){
return mid;
}
if(arr[mid]<element){
return binarysearch(arr,mid+1,right,element);
}
if(arr[mid]>element){
return binarysearch(arr,left,mid-1,element);
}
}
return -1;
}
