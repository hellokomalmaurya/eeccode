// Structure for a binary tree node
struct TreeNode {
    int data;
    struct TreeNode* left;
    struct TreeNode* right;
};

//Function to build a strictly binary tree 
from user input
struct TreeNode* buildStrictBinaryTree() {
    int data;
    printf("Enter the data for the node 
    (enter -1 to exit): ");
    scanf("%d", &data);

    if (data == -1) {
        return NULL;
    }

    struct TreeNode* root = (struct TreeNode*)
    malloc(sizeof(struct TreeNode));
    root->data = data;

    printf("Enter left child for %d:\n", data);
    root->left = buildStrictBinaryTree();

    printf("Enter right child for %d:\n", data);
    root->right = buildStrictBinaryTree();

    return root;
}

//Function to perform in-order traversal of 
   the strictly binary tree
void inorderTraversal(struct TreeNode* root) {
    if (root == NULL) {
        return;

    inorderTraversal(root->left);

    inorderTraversal(root->right);
}
