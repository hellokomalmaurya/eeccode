// Structure for a binary search tree node
struct TreeNode {
    int data;
    struct TreeNode* left;
    struct TreeNode* right;
};

int main() {
    struct TreeNode* root = NULL;
    int data, count = 0;

printf("Enter elements for the binary search tree 
(enter -1 to exit):\n");
    while (1) {
        scanf("%d", &data);
        if (data == -1) {
            break;
        }
struct TreeNode* newNode = (struct TreeNode*)
malloc(sizeof(struct TreeNode));
        newNode->data = data;
        newNode->left = NULL;
        newNode->right = NULL;

        if (root == NULL) {
            root = newNode;
        } else {
   struct TreeNode* current = root;
   struct TreeNode* parent;
   while (current != NULL) {
   parent = current;

   if (data < current->data) {
   current = current->left;
   } else {
   current = current->right;
                }
            }

   if (data < parent->data) {
   parent->left = newNode;
   } else {
   parent->right = newNode;
            }
        }

        count++;
    }

  printf("Number of nodes inserted in the BST: 
  %d\n", count);
    return 0;
}
