// Structure for a binary tree node
struct TreeNode {
    int data;
    struct TreeNode* left;
    struct TreeNode* right;
};

// Function to build a complete binary tree 
from user input
struct TreeNode* buildCompleteBinaryTree
(int* index, int n) {
    if (*index >= n) {
        return NULL;
    }

    int data;
 printf("Enter data for node %d: ", *index + 1);

        return NULL;
    }

    struct TreeNode* root = (struct TreeNode*)
    malloc(sizeof(struct TreeNode));
    root->data = data;

    (*index)++;
 root->left = buildCompleteBinaryTree(index, n);
    (*index)++;
 root->right = buildCompleteBinaryTree(index, n);

    return root;
}
