//Find the local minima in the array?
//(An element in an array is a local minima if it is less than the element after it, and the element before it)
#include<stdio.h>
#include<conio.h>
void mini(int arr[],int size){
int i;
for(i=1;i<size;i++){
if(arr[i]<arr[i-1] && arr[i]<arr[i+1]){
printf("%d",arr[i]);
}
}
}
