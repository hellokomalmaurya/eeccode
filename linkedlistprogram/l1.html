//Implement the operations of single linked list storing integer value
//(insert at begin,insert at pos,insert at end,delete at begin,delete at end,delete at pos,isempty,isfull,display,search)
#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
struct node{
int data;
struct node *next;
};
struct node* atbiggin(struct node *head,int value){
struct node *ptr=(struct node*) malloc(sizeof(struct node));
ptr->next=head;
ptr->data=value;
return ptr;
}
struct node* atpos(struct node *head,int position,int value){
struct node *ptr=(struct node*) malloc(sizeof(struct node));
struct node *p=head;
int i=1;
while(i!=position-1){
p=p->next;
i++;
}
ptr->data=value;
ptr->next=p->next;
p->next=ptr;
return head;
}
struct node* atend(struct node *head,int value){
struct node *ptr=(struct node*) malloc(sizeof(struct node));
struct node *p=head;
while(p->next!=NULL){
p=p->next;
}
ptr->data=value;
ptr->next=NULL;
p->next=ptr;
return head;
}
struct node* delatbeggin(struct node *head){
struct node *p;
p=head;
head=head->next;
free(p);
return head;
}
struct node* delatpos(struct node *head,int position){
struct node *p=head,*q;
int i=1;
while(i!=position-1){
p=p->next;
i++;
}
q=p->next;
p->next=q->next;
free(q);
return head;
}
struct node* delatend(struct node *head){
struct node *p=head,*q=head->next;
while(q->next!=NULL){
p=p->next;
q=q->next;
}
q=p->next;
p->next=NULL;
free(q);
return head;
}
void isempty(struct node *head){
if(head==NULL){
printf("List is Empty!");
}
else{
printf("List is not empty!");
}
}
void search(struct node *head,int srch){
while(head->next!=NULL){
if(head->data==srch){
printf("found!");
break;
}
head=head->next;
}
if(head->next==NULL){
printf("Not found!");
}
}
void isfull(struct node *head){
if(head!=NULL){
printf("List is full!");
}
else{
printf("Empty!");
}
}
void display(struct node *ptr){
while(ptr!=NULL){
printf("%d\t",ptr->data);
ptr=ptr->next;
}
}
