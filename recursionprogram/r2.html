#include<stdio.h>

    // Function to calculate Fibonacci using recursion
    int fibonacci(int n) {
        if (n <= 0) {
            return 0;
        } else if (n == 1) {
            return 1;
        }
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
    
    int main() {
        int num;
        printf("Enter the position in the Fibonacci sequence: ");
        scanf("%d", &num);
    
        if (num < 0) {
            printf("Fibonacci sequence is not defined for negative positions.\n");
        } else {
            int result = fibonacci(num);
            printf("Fibonacci at position %d is %d\n", num, result);
        }
    
        return 0;
    }
