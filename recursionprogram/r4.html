#include < stdio.h >

    // Function to convert a decimal number to binary using recursion
    void decimalToBinary(int n) {
        if (n > 0) {
            decimalToBinary(n / 2);
            printf("%d", n % 2);
        }
    }
    
    int main() {
        int num;
        printf("Enter a decimal number: ");
        scanf("%d", &num);
    
        if (num < 0) {
            printf("Binary representation is not defined for negative numbers.\n");
        } else if (num == 0) {
            printf("Binary representation: 0\n");
        } else {
            printf("Binary representation: ");
            decimalToBinary(num);
            printf("\n");
        }
    
        return 0;
    }
