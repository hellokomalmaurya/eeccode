#include < stdio.h >

    float calculateMean(int arr[], int size, int index, int sum) {
        if (index == size) {
            return (float)sum / size;
        } else {
            return calculateMean(arr, size, index + 1, sum + arr[index]);
        }
    }
    
    int main() {
        int size;
        printf("Enter the size of the array: ");
        scanf("%d", &size);
        
        int arr[size];
        printf("Enter the elements of the array: ");
        for (int i = 0; i < size; i++) {
            scanf("%d", &arr[i]);
        }
        
        float mean = calculateMean(arr, size, 0, 0);
        printf("The arithmetic mean of the array is: %.2f\n", mean);
        
        return 0;
    }
