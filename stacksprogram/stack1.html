int push(int stack[],int size,int item,int top){
    top++;
    stack[top]=item;
    return top;
   }

int pop(int stack[],int size,int item,int top){
    return top;
    }

void main(){
    int top=-1;
    int size,i;
    printf("Enter size : ");

    5 for isFull(),6 for IsEmpty,

    scanf("%d",&op);
    switch (op)
    {
 case 1:if(top==size-1){
         printf("Stack is full\n");}
         else{
         printf("Enter value : ");

         top = push(stack,size,item,top);

 case 2:top = pop(stack,size,item,top);

        printf("Stack is Empty\n");}

        top--;

 case 3:if(top==-1){
        printf("Stack is Empty\n");}
        else{
        top=display(stack,size,item,top);
        printf("The values in stack are : ");
        for(i=0;i<=top;i++){
        printf("%d",stack[i]);
        }
        printf("\n");
        }        
        break;
 case 4:if(top==-1){
        printf("Stack is Empty\n");}
        else{
        top=peek(stack,size,item,top);
        printf("The front value is : %d\n",
        stack[top]);
        }
        break;
 case 5:if(top==size-1){
        printf("Stack is full\n");}

        printf("Stack is not full\n");
        }
        break;
 case 6:if(top==-1){
        printf("Stack is Empty.\n");}
        else{
        printf("Stack is not Empty.\n");
        }
        break;
 case 7:printf("Exited!\n");
        exit(0);
        break;
 default:
        printf("Invalid choice!\n");
    }}}
