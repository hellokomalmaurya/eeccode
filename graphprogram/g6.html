void addEdge(int adj[], int V, int u, int v) {
    adj[u * V + v] = 1;
}

void adjacencylist(int adj[], int V) {
    for (int i = 1; i < V; i++) {
        printf("%d ->", i);
        for (int j = 1; j < V; j++) {
            if (adj[i * V + j] == 1) {
                printf(" %d", j);
            }
        }
        printf("\n");
    }
}

void BFS(int adj[], int V, int startVertex) {
    bool visited[V];
    for (int i = 0; i < V; i++) {
        visited[i] = false;
    }

    int queue[V];
    int front = -1, rear = -1;

    visited[startVertex] = true;
    queue[++rear] = startVertex;

    printf("Breadth-First Search (BFS): ");

    while (front != rear) {
        int currentVertex = queue[++front];
        printf("%d -> ", currentVertex);

        for (int i = 1; i < V; i++) {
 if (!visited[i] && adj[currentVertex * V + i] == 1) {
                visited[i] = true;
                queue[++rear] = i;
            }
        }
    }

    printf("\n");
}
