#include<stdio.h>
#include<stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Graph {
  // Number of vertices    
  int V;  
  // Array of adjacency lists
  struct Node** adjList;  
};

// Function to create a new node
struct Node* createNode(int data) {
    struct Node* newNode = (struct Node*)
   malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = NULL;
    return newNode;
}

// Function to create a graph with 'V' vertices
struct Graph* createGraph() {
    struct Graph* graph = (struct Graph*)
    malloc(sizeof(struct Graph));
    graph->V = 0;
    graph->adjList = NULL;

    return graph;
}

// Function to add a vertex to the graph
void addVertex(struct Graph* graph) {
    graph->V++;
    graph->adjList = (struct Node*)realloc
   (graph->adjList, graph->V * sizeof(struct Node));
    graph->adjList[graph->V - 1] = NULL;
}

// Function to add an edge from vertex 'src' to 
vertex 'dest'
void addEdge(struct Graph*graph,int src,int dest) {
    struct Node* newNode = createNode(dest);
    newNode->next = graph->adjList[src];
    graph->adjList[src] = newNode;
}

// Function to remove an edge from vertex 'src' to 
vertex 'dest'
void removeEdge(struct Graph*graph,int src,int dest) {
    struct Node* current = graph->adjList[src];
    struct Node* prev = NULL;

    while (current != NULL) {
    if (current->data == dest) {
      if (prev != NULL) {
       prev->next = current->next;
        } else {
     graph->adjList[src] = current->next;
            }
            free(current);
            return;
        }
        prev = current;
        current = current->next;
    }
}

// Function to print the adjacency list of the graph
void printGraph(struct Graph* graph) {
    for (int i = 0; i < graph->V; i++) {
        struct Node* current = graph->adjList[i];
        printf("Adjacency list of vertex %d: ", i);
        while (current) {
            printf("%d -> ", current->data);
            current = current->next;
        }
        printf("NULL\n");
    }
}

    struct Graph* graph = createGraph();

    int choice, src, dest;
    while (1) {
        printf("\nChoose an operation:\n");

 switch (choice) {
 case 1: addVertex(graph);

         break;
 case 2: printf("Enter source and destination: ");
         scanf("%d %d", &src, &dest);
 if (src >= 0 && src < graph->V && dest >= 0 && dest < graph->V)

 printf("Edge (%d, %d) inserted.\n", src, dest);
 } else {
 printf("Invalid vertices. Please re-enter.\n");
 }
        break;
 case 3: printf("Enter source and destination: ");

 if (src >= 0 && src < graph->V && dest >= 0 && dest < graph->V) 
 {
 removeEdge(graph, src, dest);
 printf("Edge (%d, %d) deleted.\n", src, dest);
 } else {
 printf("Invalid vertices. Please re-enter.\n");

 case 5: printf("Exiting the program.\n");
