#include<stdio.h>
#include<stdbool.h>

void addEdge(int adj[], int V, int u, int v) {
    adj[u * V + v] = 1;
}

void adjacencylist(int adj[], int V) {
    for (int i = 1; i < V; i++) {
        printf("%d ->", i);
        for (int j = 1; j < V; j++) {
            if (adj[i * V + j] == 1) {
                printf(" %d", j);
            }
        }
        printf("\n");
    }
}

void DFS(int adj[], int V, int vertex, 
bool visited[]) {
    visited[vertex] = true;
    printf("%d -> ", vertex);

 for (int i = 1; i < V; i++) {
 if (!visited[i] && adj[vertex * V + i] == 1) {
 DFS(adj, V, i, visited);
        }
    }
}

void initGraph(int V, int noOfEdges) {
    int edges[noOfEdges][2];
    int adj[V * V];

    for (int i = 0; i < V * V; i++) {
        adj[i] = 0;
    }

  for (int i = 0; i < noOfEdges; i++) {
  int source, destination;
  printf("Enter edge %d (source destination): ", 
  i + 1);
  scanf("%d %d", &source, &destination);
  if (source < 1 || source >= V || 
  destination < 1 || destination >= V) {
  printf("Invalid edge: (%d, %d) - vertices out 
  of range. Please re-enter.\n", source, 
  destination);
            i--;
        } else {
            edges[i][0] = source;
            edges[i][1] = destination;
            addEdge(adj, V, source, destination);
        }
    }

    printf("Adjacency List:\n");
    adjacencylist(adj, V);

    int startVertex;
    printf("Enter the starting vertex for DFS: ");
    scanf("%d", &startVertex);

    if (startVertex < 1 || startVertex >= V) {
    printf("Invalid starting vertex. 
    Please enter a valid vertex.\n");
    return;
    }

    bool visited[V];
    for (int i = 0; i < V; i++) {
        visited[i] = false;
    }

    printf("Depth-First Search (DFS) starting 
    from vertex %d: ", startVertex);
    DFS(adj, V, startVertex, visited);
    printf("\n");
}

int main() {
    int V;
    printf("Enter the number of vertices: ");
    scanf("%d", &V);

    int noOfEdges;
    printf("Enter the number of edges: ");
    scanf("%d", &noOfEdges);

    initGraph(V, noOfEdges);

    return 0;
}
